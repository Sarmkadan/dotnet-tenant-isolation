using System.Collections.Generic;
using System.ComponentModel.DataAnnotations;
using System.Linq;

namespace TenantIsolation.Configuration
{
    /// <summary>
    /// Strongly‑typed configuration options for the Tenant Isolation library.
    /// These options are bound from the "TenantIsolation" section of appsettings.json
    /// and validated using DataAnnotations (and a custom validation rule).
    /// </summary>
    public class DotnetTenantIsolationOptions : IValidatableObject
    {
        /// <summary>
        /// The default isolation strategy to use when a tenant does not specify one.
        /// </summary>
        [Required]
        public string DefaultStrategy { get; set; } = "DatabasePerTenant";

        /// <summary>
        /// The list of isolation strategies that are allowed by the application.
        /// </summary>
        [Required]
        public List<string> EnabledStrategies { get; set; } = new List<string>();

        /// <summary>
        /// Maximum number of tenants that can be created on a single instance.
        /// </summary>
        [Range(1, int.MaxValue, ErrorMessage = "MaxTenantsPerInstance must be greater than zero.")]
        public int MaxTenantsPerInstance { get; set; } = 1000;

        /// <summary>
        /// Settings related to caching of tenant data.
        /// </summary>
        public CachingOptions Caching { get; set; } = new CachingOptions();

        /// <summary>
        /// Settings for building connection strings for tenant databases.
        /// </summary>
        public ConnectionStringsOptions ConnectionStrings { get; set; } = new ConnectionStringsOptions();

        public class CachingOptions
        {
            /// <summary>
            /// Enables or disables the internal cache.
            /// </summary>
            public bool Enabled { get; set; } = true;

            /// <summary>
            /// Cache duration in seconds.
            /// </summary>
            [Range(1, int.MaxValue, ErrorMessage = "Caching.Duration must be greater than zero.")]
            public int Duration { get; set; } = 3600;
        }

        public class ConnectionStringsOptions
        {
            /// <summary>
            /// Template used to build a tenant‑specific connection string.
            /// Placeholders {server} and {database} will be replaced at runtime.
            /// </summary>
            [Required]
            public string Template { get; set; } = "Server={server};Database={database};Trusted_Connection=true;TrustServerCertificate=true;";
        }

        /// <summary>
        /// Custom validation logic.
        /// Ensures that the default strategy is part of the enabled strategies list.
        /// </summary>
        public IEnumerable<ValidationResult> Validate(ValidationContext validationContext)
        {
            if (EnabledStrategies == null || !EnabledStrategies.Any())
            {
                yield return new ValidationResult(
                    "EnabledStrategies must contain at least one strategy.",
                    new[] { nameof(EnabledStrategies) });
                yield break;
            }

            if (!EnabledStrategies.Contains(DefaultStrategy))
            {
                yield return new ValidationResult(
                    $"EnabledStrategies must contain the DefaultStrategy value '{DefaultStrategy}'.",
                    new[] { nameof(EnabledStrategies), nameof(DefaultStrategy) });
            }
        }
    }
}
